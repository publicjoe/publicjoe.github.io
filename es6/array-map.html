Array map

The JavaScript Array type provides the map() method that allows you to transform array elements in a clean way.

Example

let circles = [
    10, 30, 50
];

function circleArea(radius) {
    return Math.floor(Math.PI * radius * radius);
}
let areas = circles.map(circleArea);
console.log(areas); // Output: [ 314, 2827, 7853 ]

In ES6, this code looks like this in an arrow function

let circles = [
    10, 30, 50
];

let areas = circles.map(radius => Math.floor(Math.PI * radius * radius));
console.log(areas); // Output: [ 314, 2827, 7853 ]
