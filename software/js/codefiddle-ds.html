1. HTML Structure (index.html)

This file creates the basic layout with tabs, editor containers, and preview panel.

index.html 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <title>CodeFiddle - Web Code Editor</title>
  <link rel="stylesheet" href="styles.css">
</head>
<body>
  <div id="container">
    <!-- Editor Panel -->
    <div id="editor-panel">
      <div id="tabs">
        <button data-tab="html" class="active">HTML</button>
        <button data-tab="css">CSS</button>
        <button data-tab="js">JavaScript</button>
        <button data-tab="ts">TypeScript</button>
      </div>

      <!-- Editor Containers -->
      <div id="html-editor" class="editor-container active"></div>
      <div id="css-editor" class="editor-container"></div>
      <div id="js-editor" class="editor-container"></div>
      <div id="ts-editor" class="editor-container"></div>
    </div>

    <!-- Preview Panel -->
    <div id="preview-panel">
      <iframe id="preview-frame" sandbox="allow-scripts allow-same-origin"></iframe>
    </div>
  </div>

  <!-- Monaco Editor from CDN -->
  <script src="https://cdn.jsdelivr.net/npm/monaco-editor@0.47.0/min/vs/loader.js"></script>
  <script src="script.js"></script>
</body>
</html>

What This HTML Does:

  • Basic Structure: Sets up a standard HTML5 document
  • Container: Creates a flex container to hold both editor and preview panels
  • Tabs: Navigation buttons to switch between different code types
  • Editor Containers: Empty divs where Monaco Editor will initialize
  • Preview Frame: Iframe that shows the live output
  • Scripts: Loads Monaco Editor from CDN and our custom JavaScript

3. JavaScript Functionality (script.js)

This file initializes Monaco Editor, handles tab switching, and manages the live preview.

script.js 
// Global object to store all editor instances
let editors = {};

// Configure and load Monaco Editor
require.config({ 
  paths: { 
    'vs': 'https://cdn.jsdelivr.net/npm/monaco-editor@0.47.0/min/vs' 
  } 
});

// Initialize Monaco Editor when the page loads
require(['vs/editor/editor.main'], () => {
  initializeEditors();
  setupTabSystem();
  setupLivePreview();
});

function initializeEditors() {
  // Create HTML Editor
  editors.html = monaco.editor.create(document.getElementById('html-editor'), {
    value: `<!DOCTYPE html>
<html>
<head>
    <title>My Page</title>
</head>
<body>
    <h1>Welcome to CodeFiddle!</h1>
    <p>Start editing to see live changes.</p>
    <div id="app"></div>
</body>
</html>`,
    language: 'html',
    theme: 'vs-dark',
    automaticLayout: true,
    fontSize: 14,
    lineNumbers: 'on'
  });

  // Create CSS Editor
  editors.css = monaco.editor.create(document.getElementById('css-editor'), {
    value: `body {
    font-family: Arial, sans-serif;
    margin: 40px;
    background: #f0f0f0;
}

h1 {
    color: #007acc;
    text-align: center;
}

p {
    color: #333;
    line-height: 1.6;
}`,
    language: 'css',
    theme: 'vs-dark',
    automaticLayout: true,
    fontSize: 14,
    lineNumbers: 'on'
  });

  // Create JavaScript Editor
  editors.js = monaco.editor.create(document.getElementById('js-editor'), {
    value: `// JavaScript code here
console.log('Hello from JavaScript!');

// Example: Update content dynamically
document.addEventListener('DOMContentLoaded', function() {
    const app = document.getElementById('app');
    if (app) {
        app.innerHTML = '<p>This content was added by JavaScript!</p>';
    }
});`,
    language: 'javascript',
    theme: 'vs-dark',
    automaticLayout: true,
    fontSize: 14,
    lineNumbers: 'on'
  });

  // Create TypeScript Editor
  editors.ts = monaco.editor.create(document.getElementById('ts-editor'), {
    value: `// TypeScript code here
interface Message {
    text: string;
    timestamp: Date;
}

const message: Message = {
    text: 'Hello from TypeScript!',
    timestamp: new Date()
};

console.log(message.text);

// Example function with TypeScript
function multiply(a: number, b: number): number {
    return a * b;
}

const result = multiply(5, 3);
console.log('5 * 3 =', result);`,
    language: 'typescript',
    theme: 'vs-dark',
    automaticLayout: true,
    fontSize: 14,
    lineNumbers: 'on'
  });
}

function setupTabSystem() {
  // Add click event listeners to all tab buttons
  document.querySelectorAll('#tabs button').forEach(btn => {
    btn.addEventListener('click', () => {
      // Remove active class from all tabs
      document.querySelectorAll('#tabs button.active').forEach(activeBtn => {
        activeBtn.classList.remove('active');
      });
      
      // Add active class to clicked tab
      btn.classList.add('active');
      
      // Hide all editor containers
      document.querySelectorAll('.editor-container.active').forEach(container => {
        container.classList.remove('active');
      });
      
      // Show the selected editor
      const tabName = btn.dataset.tab;
      document.getElementById(tabName + '-editor').classList.add('active');
    });
  });
}

function setupLivePreview() {
  // Create debounced update function to prevent excessive recompilation
  const debouncedUpdate = debounce(updatePreview, 500);
  
  // Set up change listeners for all editors
  Object.values(editors).forEach(editor => {
    editor.onDidChangeModelContent(debouncedUpdate);
  });
  
  // Initial preview render
  updatePreview();
}

function updatePreview() {
  // Get current code from all editors
  const html = editors.html.getValue();
  const css = `<style>${editors.css.getValue()}</style>`;
  const js = editors.js.getValue();
  
  // Compile TypeScript to JavaScript
  compileTypeScript().then(compiledJs => {
    // Combine all code for the preview
    const combinedCode = html + css + `<script>${js}\n${compiledJs}</script>`;
    
    // Update the iframe
    const iframe = document.getElementById('preview-frame');
    iframe.srcdoc = combinedCode;
  });
}

function compileTypeScript() {
  return new Promise((resolve) => {
    // Use Monaco's TypeScript compiler to transpile TS to JS
    monaco.languages.typescript.getTypeScriptWorker()
      .then(worker => worker(editors.ts.getModel().uri))
      .then(client => client.getEmitOutput(editors.ts.getModel().uri.toString()))
      .then(output => {
        const compiledCode = output.outputFiles[0]?.text || "";
        resolve(compiledCode);
      })
      .catch(error => {
        console.error('TypeScript compilation error:', error);
        resolve('// TypeScript compilation failed');
      });
  });
}

// Utility function to debounce rapid updates
function debounce(func, wait) {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
}

What This JavaScript Does:

  • Monaco Initialization: Loads the Monaco Editor from Microsoft's CDN
  • Editor Creation: Sets up four code editors with syntax highlighting
  • Tab Management: Handles switching between different code types
  • Live Preview: Automatically updates the preview when code changes
  • TypeScript Compilation: Compiles TypeScript to JavaScript in real-time
  • Debouncing: Prevents excessive updates while typing

Key Features Explained

Monaco Editor Integration

We use Microsoft's Monaco Editor (the same engine that powers VS Code) for professional-grade code editing with syntax highlighting, line numbers, auto-completion, and error checking.

TypeScript Support

The editor automatically compiles TypeScript to JavaScript using Monaco's built-in TypeScript compiler, making it perfect for learning TypeScript.

Live Preview System

The preview updates automatically thanks to event listeners on editor content changes, debouncing to prevent excessive updates, and dynamic iframe content replacement.

Security

The sandbox="allow-scripts allow-same-origin" attribute on the iframe provides security isolation while allowing scripts to run.
