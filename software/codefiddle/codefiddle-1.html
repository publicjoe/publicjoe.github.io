1. HTML Structure (index.html)

This file creates the basic layout with tabs, editor containers, and preview panel.

index.html 
<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="UTF-8" />
<title>Mini Code Editor with Monaco & TypeScript</title>
<link rel="stylesheet" href="style.css">
</head>
<body>
<div id="container">
<div id="editor-panel">
    <div id="tabs">
    <button data-tab="html" class="active">HTML</button>
    <button data-tab="css">CSS</button>
    <button data-tab="js">JavaScript</button>
    <button data-tab="ts">TypeScript</button>
    </div>

    <div id="html-editor" class="editor-container active"></div>
    <div id="css-editor" class="editor-container"></div>
    <div id="js-editor" class="editor-container"></div>
    <div id="ts-editor" class="editor-container"></div>
</div>

<div id="preview-panel">
    <iframe id="preview-frame" sandbox="allow-scripts allow-same-origin"></iframe>
</div>
</div>

<!-- Monaco Editor Loader -->
<script src="https://cdn.jsdelivr.net/npm/monaco-editor@0.47.0/min/vs/loader.js"></script>

<script src="app.js" defer></script>
</body>
</html>

What This HTML Does:

  • Basic Structure: Sets up a standard HTML5 document
  • Container: Creates a flex container to hold both editor and preview panels
  • Tabs: Navigation buttons to switch between different code types
  • Editor Containers: Empty divs where Monaco Editor will initialize
  • Preview Frame: Iframe that shows the live output
  • Scripts: Loads Monaco Editor from CDN and our custom JavaScript, a point to note, is the Monaco script needs time to load, so we use the defer keyword to do this.

3. JavaScript Functionality (script.js)

This file initializes Monaco Editor, handles tab switching, and manages the live preview.

script.js 
// Global object to store all editor instances
let editors = {};

require.config({ paths: { 'vs': 'https://cdn.jsdelivr.net/npm/monaco-editor@0.47.0/min/vs' } });
require(['vs/editor/editor.main'], () => {

// --- Create editors ---
editors.html = monaco.editor.create(document.getElementById('html-editor'), {
    value: "<h1>Welcome to CodeFiddle!</h1>\n<p>Start editing to see live changes.</p>",
    language: 'html',
    theme: 'vs-dark',
    automaticLayout: true
});

editors.css = monaco.editor.create(document.getElementById('css-editor'), {
    value: "h1 { color: dodgerblue; }",
    language: 'css',
    theme: 'vs-dark',
    automaticLayout: true
});

editors.js = monaco.editor.create(document.getElementById('js-editor'), {
    value: "console.log('Hello from JavaScript');",
    language: 'javascript',
    theme: 'vs-dark',
    automaticLayout: true
});

editors.ts = monaco.editor.create(document.getElementById('ts-editor'), {
    value: "let message: string = 'Hello from TypeScript';\nconsole.log(message);",
    language: 'typescript',
    theme: 'vs-dark',
    automaticLayout: true
});

// --- Tab switching ---
document.querySelectorAll('#tabs button').forEach(btn => {
    btn.addEventListener('click', () => {
        document.querySelector('#tabs button.active').classList.remove('active');
        btn.classList.add('active');

        document.querySelector('.editor-container.active').classList.remove('active');
        document.getElementById(btn.dataset.tab + '-editor').classList.add('active');
    });
});

// --- Preview rendering ---
function updatePreview() {
const html = editors.html.getValue();
const css = `<style>${editors.css.getValue()}</style>`;
const js = editors.js.getValue();

// Compile TypeScript to JavaScript using Monaco's built-in TS compiler
const tsCode = editors.ts.getValue();
const transpiled = monaco.languages.typescript.getTypeScriptWorker()
    .then(worker => worker(editors.ts.getModel().uri))
    .then(client => client.getEmitOutput(editors.ts.getModel().uri.toString()))
    .then(output => {
        const tsJs = output.outputFiles[0]?.text || "";
        const combined = html + css + `<script>${js}\n${tsJs}<\/script>`;
        const iframe = document.getElementById('preview-frame');
        iframe.srcdoc = combined;
    });
}

// --- Debounce updates to avoid excessive recompiles ---
function debounce(fn, delay = 400) {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => fn(...args), delay);
    };
}

const debouncedUpdate = debounce(updatePreview);

// Trigger preview updates on change
Object.values(editors).forEach(editor => {
    editor.onDidChangeModelContent(debouncedUpdate);
});

// Initial render
updatePreview();
});

What This JavaScript Does:

  • Monaco Initialization: Loads the Monaco Editor from Microsoft's CDN
  • Editor Creation: Sets up four code editors with syntax highlighting
  • Tab Management: Handles switching between different code types
  • Live Preview: Automatically updates the preview when code changes
  • TypeScript Compilation: Compiles TypeScript to JavaScript in real-time
  • Debouncing: Prevents excessive updates while typing

Key Features Explained

Monaco Editor Integration

We use Microsoft's Monaco Editor (the same engine that powers VS Code) for professional-grade code editing with syntax highlighting, line numbers, auto-completion, and error checking.

TypeScript Support

The editor automatically compiles TypeScript to JavaScript using Monaco's built-in TypeScript compiler, making it perfect for learning TypeScript.

Live Preview System

The preview updates automatically thanks to event listeners on editor content changes, debouncing to prevent excessive updates, and dynamic iframe content replacement.

Security

The sandbox="allow-scripts allow-same-origin" attribute on the iframe provides security isolation while allowing scripts to run.
